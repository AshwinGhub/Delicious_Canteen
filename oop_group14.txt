#include<iostream>
#include<string>
using namespace std;
class point
{
	int x,y;
	public:
		point()
		{
			x=0;y=0;
		}
		void set(int a,int b)
		{
			x=a; y=b;
		}
		int getx()
		{
			return(x);
		}
		int gety()
		{
			return y;
		}
};
class rectangle
{
	point tleft;
	point bright;
	public : 
		void get_coordinate(int p,int q,int r, int s)
		{
			tleft.set(p,q);
			bright.set(r,s);
		}
		void disp_coordinates()
		{
			cout<<"Top left : ";
			cout<<tleft.getx()<<" "<<tleft.gety()<<endl;
			cout<<"Bottom right : ";cout<<bright.getx()<<" "<<bright.gety()<<endl;
		}
		int perimeter()
		{
			int l=bright.getx()-tleft.getx();
			int w=tleft.gety()-bright.gety();
			return(2*(l+w));
		}
};
void sort(rectangle arr[],int a)
{
	rectangle temp;
	for(int i=0;i<a;i++)
	 {
	 	for(int j=0;j<a-1;j++)
	    if(arr[i].perimeter()<=arr[i+1].perimeter())
		{
		 temp=arr[i];
		 arr[i]=arr[i+1];
		 arr[i+1]=temp;
     }}
     for(int i=0;i<a;i++)
     {
     	arr[i].disp_coordinates();
     	cout<<"Perimeter = "<<arr[i].perimeter()<<endl<<endl;
	 }
}
int main()
{
	int p,q,r,s; int a;
	cout<<"Enter number of rectangles : "; cin>>a;
	rectangle rect[a];
	rectangle *rptr=rect;
	for(int i=0;i<a;i++)
	{ 
		cout<<"Enter top left point of rectangle "<<i+1<<" : ";
		cin>>p>>q;
		cout<<"Enter bottom right point of rectangle "<<i+1<<" : ";
		cin>>r>>s;
		rptr->get_coordinate(p,q,r,s);
		rptr++; 
	}
	rptr=&rect[0];
	sort(rect,a);
	return 0;
}
